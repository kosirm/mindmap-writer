HTML:
<div id="q-app">
  <q-layout view="hHh Lpr lFf">
		<q-header class="bg-primary">
			<q-toolbar>
				<q-btn flat round icon="menu" @click="drawerLeft = !drawerLeft"></q-btn>
				<q-toolbar-title>Test app</q-toolbar-title>
			</q-toolbar>
		</q-header>

		<q-splitter v-model="splitterModel" unit="px" emit-immediately :class="splitterClass" @input="onChange">
			<template #before>
				<q-drawer v-model="drawerLeft" :width="drawerWidth" content-class="bg-grey-5" :breakpoint="0">
<!-- breakpoint is needed or force behavior to desktop otherwise on small screen will enforce an overlay because it will go to mobile mode which is the normal behavior -->
					<q-scroll-area class="fit">
						<div class="q-pa-sm">
							<div v-for="n in 50" :key="n">
								Drawer {{ n }} / 50
							</div>
						</div>
					</q-scroll-area>
				</q-drawer>
			</template>

			<template #after>
				<q-page-container :style="compPadding">
					<q-page padding>
						<q-checkbox v-model="booleanTest" label="Checkbox" ></q-checkbox>
					</q-page>
				</q-page-container>
			</template>
		</q-splitter>
	</q-layout>
</div>

CSS:
.splitt
  .q-splitter__before
    transition: width 0.2s ease-out
    width: 0px !important

JAVASCRIPT:
new Vue({
  el: '#q-app',
	data() {
		return {
			booleanTest: false,
			drawerLeft: true,
      drawerRight: false,
			splitterModel: 200
		};
	},
  computed: {
    drawerWidth() {
      return this.splitterModel +1
    },
    splitterClass() {
      return !this.drawerLeft ? 'splitt': ''
    },
    compPadding() {
      return this.drawerLeft ? { paddingLeft: '0px' } : ''
    }
  },
  methods: {
    onChange(v) {
      this.drawerLeft = v > 0
    }
  }
});
